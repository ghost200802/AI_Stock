# 2. 蔡森形态判断

## 一、模块定位

蔡森形态识别模块位于 **D. 股价形态判断层**，属于项目核心业务逻辑，放在 `modules/caisen/` 下。基于缠论模块输出的笔序列，识别蔡森理论定义的12种核心形态，进行强弱评分、目标价计算和量价验证。所有识别结果通过 `lib/db/` 持久化到 MongoDB。

**依赖关系**：
- 笔序列输入：来自 `modules/chanlun/` 的 `List[Bi]`
- K线数据输入：来自 `lib/data_source/` 的 DataFrame
- 量价验证：来自 `modules/volume_price/` 的验证评分
- 缓存读写：通过 `lib/db/cache_manager.py` 操作 MongoDB

## 二、数据流

```
modules/chanlun/ : List[Bi] (笔序列)
    ↓
modules/caisen/pattern_scanner.py
    ↓ 各形态识别器并行检测
    ├─ modules/caisen/w_bottom.py
    ├─ modules/caisen/m_top.py
    ├─ modules/caisen/head_shoulder_bottom.py
    ├─ modules/caisen/head_shoulder_top.py
    ├─ modules/caisen/breakout_fail.py
    ├─ modules/caisen/podie_fan.py
    ├─ modules/caisen/flag_down.py
    ├─ modules/caisen/flag_up.py
    └─ modules/caisen/triangle.py
    ↓
List[Pattern] (识别到的形态列表)
    ↓
modules/caisen/strength_scorer.py
    ↓ 评分 + modules/volume_price/volume_verifier.py 量价验证
Pattern (含 strength_score, volume_score, target_price, status)
    ↓ 通过 lib/db/cache_manager.py 缓存到 MongoDB
```

from enum import Enum
from dataclasses import dataclass, field
from typing import List, Dict, Optional, Any
from datetime import datetime

class PatternType(Enum):
    W_BOTTOM = "w_bottom"                    # W底（双重底）
    M_TOP = "m_top"                          # M头（双重顶）
    HEAD_SHOULDER_BOTTOM = "head_shoulder_bottom"  # 头肩底
    HEAD_SHOULDER_TOP = "head_shoulder_top"        # 头肩顶
    BREAKOUT_FAIL = "breakout_fail"          # 假突破
    PODIE_FAN = "podie_fan"                  # 破底翻
    FLAG_DOWN = "flag_down"                  # 下飘旗形
    FLAG_UP = "flag_up"                      # 上飘旗形
    TRIANGLE_HEAD = "triangle_head"          # 头部收敛三角形
    TRIANGLE_BOTTOM = "triangle_bottom"      # 底部收敛三角形

class PatternStatus(Enum):
    FORMING = "forming"        # 形成中（颈线尚未突破）
    CONFIRMED = "confirmed"    # 已确认（颈线已突破且满足量价条件）
    FAILED = "failed"          # 已失效（形态破坏）

class PatternDirection(Enum):
    BULLISH = "bullish"        # 看涨形态（底部反转、下飘旗形）
    BEARISH = "bearish"        # 看跌形态（顶部反转、上飘旗形）

@dataclass
class KeyPoint:
    name: str                  # 关键点名称（如 "left_bottom", "neckline"）
    price: float               # 价格
    date: str                  # 日期

@dataclass
class PatternResult:
    ts_code: str
    period: str
    pattern_type: PatternType
    direction: PatternDirection
    status: PatternStatus
    start_date: str
    end_date: str
    neckline_price: float
    key_points: Dict[str, KeyPoint]  # 各关键点位
    strength_score: float = 0.0      # 强弱评分 (0-100)
    volume_score: float = 0.0        # 量价验证得分 (0-100)
    target_price: Optional[float] = None  # 目标价
    stop_loss_price: Optional[float] = None  # 止损价
    bi_indices: List[int] = field(default_factory=list)  # 构成形态的笔索引
    confidence: float = 0.0          # 综合置信度
    update_time: str = ""

### 3.3 形态识别器基类
```python
from abc import ABC, abstractmethod
from modules.chanlun import Bi

class BasePatternDetector(ABC):
    """
    所有形态识别器的基类
    每个具体形态继承此类，实现 detect 方法
    """

    def __init__(self, config: dict):
        self.config = config

    @abstractmethod
    def get_pattern_type(self) -> PatternType:
        """返回该识别器对应的形态类型"""
        pass

    @abstractmethod
    def get_min_bi_count(self) -> int:
        """该形态最少需要的笔数量"""
        pass

    @abstractmethod
    def detect(self, bis: List[Bi], kline_df: pd.DataFrame) -> List[PatternResult]:
        """
        在笔序列中检测形态
        Args:
            bis: 笔序列（来自 modules/chanlun）
            kline_df: 原始K线数据（来自 lib/data_source）
        Returns:
            检测到的形态列表
        """
        pass

    def _check_bi_window(self, bis: List[Bi], start_idx: int) -> bool:
        """检查从 start_idx 开始是否有足够的笔"""
        return start_idx + self.get_min_bi_count() <= len(bis)
```

## 四、各形态识别算法

### 4.1 W底（双重底）

#### 文件：`modules/caisen/w_bottom.py`

#### 形态结构
需要 **4 笔** 完成闭环，笔方向：`down → up → down → up`

```
价格
  │    B1(高点)
  │   / \
  │  /   \
  │ /     \
  │A1     A2(右底)  ← A1和A2价格差 ≤ 5%
  │       /
  │      /
  │     C(突破点，高于B1)
  └──────────────→ 时间
```

#### 识别算法
```python
class WBottomDetector(BasePatternDetector):

    def get_pattern_type(self) -> PatternType:
        return PatternType.W_BOTTOM

    def get_min_bi_count(self) -> int:
        return 4

    def detect(self, bis: List[Bi], kline_df: pd.DataFrame) -> List[PatternResult]:
        results = []
        tolerance = self.config.get("caisen", {}).get("w_bottom_tolerance", 0.05)

        for i in range(len(bis) - 3):
            b0, b1, b2, b3 = bis[i], bis[i+1], bis[i+2], bis[i+3]

            if not (b0.direction == "down" and b1.direction == "up" and
                    b2.direction == "down" and b3.direction == "up"):
                continue

            left_bottom = b0.end_price
            right_bottom = b2.end_price
            bottom_diff = abs(left_bottom - right_bottom) / min(left_bottom, right_bottom)
            if bottom_diff > tolerance:
                continue

            neckline = b1.end_price
            if b3.end_price <= neckline:
                status = PatternStatus.FORMING
            else:
                status = PatternStatus.CONFIRMED

            pattern = PatternResult(
                ts_code="", period="",
                pattern_type=PatternType.W_BOTTOM,
                direction=PatternDirection.BULLISH,
                status=status,
                start_date=b0.start_date,
                end_date=b3.end_date,
                neckline_price=neckline,
                key_points={
                    "left_bottom": KeyPoint("left_bottom", left_bottom, b0.end_date),
                    "neckline_left": KeyPoint("neckline_left", neckline, b1.end_date),
                    "right_bottom": KeyPoint("right_bottom", right_bottom, b2.end_date),
                    "breakout": KeyPoint("breakout", b3.end_price, b3.end_date),
                },
                bi_indices=list(range(i, i + 4)),
            )
            results.append(pattern)

        return results
```

### 4.3 头肩底

#### 文件：`modules/caisen/head_shoulder_bottom.py`

#### 形态结构
需要 **6 笔**，笔方向：`down → up → down → up → down → up`

```
价格
  │   B1(左肩高点)      B3(右肩高点)
  │  / \              / \
  │ /   \            /   \
  │/     \    /    /     \
  │A1(左肩) A2(头部) A3(右肩)
  │        (最低点)
  └──────────────────────→ 时间
```

#### 识别算法
```python
class HeadShoulderBottomDetector(BasePatternDetector):

    def get_pattern_type(self) -> PatternType:
        return PatternType.HEAD_SHOULDER_BOTTOM

    def get_min_bi_count(self) -> int:
        return 6

    def detect(self, bis: List[Bi], kline_df: pd.DataFrame) -> List[PatternResult]:
        results = []
        tolerance = self.config.get("caisen", {}).get("head_shoulder_tolerance", 0.10)

        for i in range(len(bis) - 5):
            b = bis[i:i+6]
            directions = [bi.direction for bi in b]
            if directions != ["down", "up", "down", "up", "down", "up"]:
                continue

            left_shoulder = b[0].end_price
            head = b[2].end_price
            right_shoulder = b[4].end_price

            if head >= left_shoulder or head >= right_shoulder:
                continue

            shoulder_diff = abs(left_shoulder - right_shoulder) / min(left_shoulder, right_shoulder)
            if shoulder_diff > tolerance:
                continue

            neckline = (b[1].end_price + b[3].end_price) / 2

            if b[5].end_price <= neckline:
                status = PatternStatus.FORMING
            else:
                status = PatternStatus.CONFIRMED

            pattern = PatternResult(
                ts_code="", period="",
                pattern_type=PatternType.HEAD_SHOULDER_BOTTOM,
                direction=PatternDirection.BULLISH,
                status=status,
                start_date=b[0].start_date,
                end_date=b[5].end_date,
                neckline_price=neckline,
                key_points={
                    "left_shoulder": KeyPoint("left_shoulder", left_shoulder, b[0].end_date),
                    "head": KeyPoint("head", head, b[2].end_date),
                    "right_shoulder": KeyPoint("right_shoulder", right_shoulder, b[4].end_date),
                    "neckline": KeyPoint("neckline", neckline, b[5].end_date),
                },
                bi_indices=list(range(i, i + 6)),
            )
            results.append(pattern)

        return results
```

## 五、形态扫描器

### 5.1 文件：`modules/caisen/pattern_scanner.py`

```python
from modules.chanlun import Bi

class PatternScanner:
    """
    统一形态扫描入口
    并行运行所有形态识别器，汇总结果
    """

    def __init__(self, config: dict):
        self.config = config
        self.detectors = [
            WBottomDetector(config),
            MTopDetector(config),
            HeadShoulderBottomDetector(config),
            HeadShoulderTopDetector(config),
            BreakoutFailDetector(config),
            PoDieFanDetector(config),
            FlagDownDetector(config),
            FlagUpDetector(config),
            TriangleDetector(config),
        ]

    def scan(self, ts_code: str, period: str,
             bis: List[Bi], kline_df: pd.DataFrame) -> List[PatternResult]:
        """
        扫描所有形态
        """
        all_patterns = []

        with ThreadPoolExecutor(max_workers=len(self.detectors)) as executor:
            futures = {
                executor.submit(detector.detect, bis, kline_df): detector
                for detector in self.detectors
            }
            for future in as_completed(futures):
                patterns = future.result()
                for p in patterns:
                    p.ts_code = ts_code
                    p.period = period
                all_patterns.extend(patterns)

        all_patterns = self._deduplicate(all_patterns)
        return all_patterns

    def _deduplicate(self, patterns: List[PatternResult]) -> List[PatternResult]:
        """
        去重逻辑：时间重叠的形态中，保留置信度最高的
        """
        if not patterns:
            return []

        patterns.sort(key=lambda p: (p.start_date, -p.strength_score - p.volume_score))
        result = []
        last_end = ""

        for p in patterns:
            if p.start_date > last_end:
                result.append(p)
                last_end = p.end_date

        return result
```

#### W底量价验证
```python
def score_w_bottom_volume(self, pattern: PatternResult, kline_df: pd.DataFrame) -> float:
    score = 0.0

    left_vol = self._get_avg_volume(kline_df, pattern.key_points["left_bottom"].date, 5)
    right_vol = self._get_avg_volume(kline_df, pattern.key_points["right_bottom"].date, 5)
    if right_vol < left_vol:
        score += 10 * (1 - right_vol / left_vol)

    if pattern.status == PatternStatus.CONFIRMED:
        breakout_vol = self._get_volume(kline_df, pattern.key_points["breakout"].date)
        ma5_vol = self._get_ma_volume(kline_df, pattern.key_points["breakout"].date, 5)
        ratio = breakout_vol / ma5_vol if ma5_vol > 0 else 0
        if ratio > 1.5:
            score += 10

    bottom_vol = max(left_vol, right_vol)
    top_vol = self._get_avg_volume(kline_df, pattern.key_points["neckline_left"].date, 5)
    if top_vol > 0 and bottom_vol / top_vol >= 0.80:
        score += 10

    return min(score, 30.0)
```

### 6.4 综合评分实现
```python
from modules.volume_price.volume_verifier import VolumeVerifier

class StrengthScorer:
    def __init__(self, config: dict):
        self.config = config
        self.volume_verifier = VolumeVerifier(config)

    def score(self, pattern: PatternResult, kline_df: pd.DataFrame) -> PatternResult:
        """
        对形态进行综合评分
        """
        pattern.volume_score = self.volume_verifier.verify(pattern, kline_df)
        breakthrough_score = self._score_breakthrough(pattern)
        time_score = self._score_time(pattern)
        position_score = self._score_position(pattern, kline_df)

        pattern.strength_score = (
            pattern.volume_score +
            breakthrough_score +
            time_score +
            position_score
        )

        pattern.confidence = (pattern.strength_score + pattern.volume_score) / 2
        return pattern
```

## 七、目标价计算

### 7.1 文件：`modules/caisen/target_price.py`

### 7.2 等幅测距原则

```python
class TargetPriceCalculator:

    def calculate(self, pattern: PatternResult) -> PatternResult:
        """
        根据形态类型计算目标价
        """
        calculator = {
            PatternType.W_BOTTOM: self._calc_w_bottom_target,
            PatternType.M_TOP: self._calc_m_top_target,
            PatternType.HEAD_SHOULDER_BOTTOM: self._calc_hs_bottom_target,
            PatternType.HEAD_SHOULDER_TOP: self._calc_hs_top_target,
            PatternType.FLAG_DOWN: self._calc_flag_down_target,
            PatternType.FLAG_UP: self._calc_flag_up_target,
            PatternType.TRIANGLE_BOTTOM: self._calc_triangle_bottom_target,
            PatternType.TRIANGLE_HEAD: self._calc_triangle_head_target,
        }

        func = calculator.get(pattern.pattern_type)
        if func:
            pattern.target_price = func(pattern)
            pattern.stop_loss_price = self._calc_stop_loss(pattern)

        return pattern

    def _calc_w_bottom_target(self, pattern: PatternResult) -> float:
        bottom = min(
            pattern.key_points["left_bottom"].price,
            pattern.key_points["right_bottom"].price
        )
        height = pattern.neckline_price - bottom
        return pattern.neckline_price + height

    def _calc_m_top_target(self, pattern: PatternResult) -> float:
        top = max(
            pattern.key_points["left_top"].price,
            pattern.key_points["right_top"].price
        )
        height = top - pattern.neckline_price
        return pattern.neckline_price - height

    def _calc_hs_bottom_target(self, pattern: PatternResult) -> float:
        head = pattern.key_points["head"].price
        height = pattern.neckline_price - head
        return pattern.neckline_price + 1.382 * height

    def _calc_hs_top_target(self, pattern: PatternResult) -> float:
        head = pattern.key_points["head"].price
        height = head - pattern.neckline_price
        return pattern.neckline_price - 1.382 * height

    def _calc_flag_down_target(self, pattern: PatternResult) -> float:
        height = max(kp.price for kp in pattern.key_points.values()) - \
                 min(kp.price for kp in pattern.key_points.values())
        return pattern.key_points["breakout"].price + height

    def _calc_stop_loss(self, pattern: PatternResult) -> float:
        if pattern.direction == PatternDirection.BULLISH:
            lowest = min(kp.price for kp in pattern.key_points.values())
            return lowest * 0.97
        else:
            highest = max(kp.price for kp in pattern.key_points.values())
            return highest * 1.03
```

## 八、缓存策略

### 8.1 缓存规则
- 形态识别结果持久化到 MongoDB `caisen_patterns` collection
- 唯一键：`{ts_code, period, pattern_type, start_date}`
- 通过 `lib/db/cache_manager.py` 进行缓存读写
- 增量更新：新增K线后，删除 `status != "confirmed"` 的旧记录，重新扫描

### 8.2 缓存优先读取
```python
from lib.db.cache_manager import CacheManager

def scan_with_cache(self, ts_code: str, period: str,
                    bis: List[Bi], kline_df: pd.DataFrame,
                    cache_manager: CacheManager) -> List[PatternResult]:
    """
    优先从缓存读取已确认形态，只对新数据重新扫描
    """
    cached = cache_manager.find_analysis_cache("caisen_patterns", {
        "ts_code": ts_code,
        "period": period,
        "status": "confirmed"
    })

    if not bis:
        return self._df_to_patterns(cached)

    latest_bi_date = bis[-1].end_date
    latest_cached_date = cached["end_date"].max() if not cached.empty else ""

    if latest_bi_date <= latest_cached_date:
        return self._df_to_patterns(cached)

    new_patterns = self.scan(ts_code, period, bis, kline_df)
    all_patterns = self._df_to_patterns(cached) + new_patterns
    all_patterns = self._deduplicate(all_patterns)
    self._save_patterns(all_patterns, cache_manager)

    return all_patterns
```
