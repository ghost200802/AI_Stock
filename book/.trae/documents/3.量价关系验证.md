# 3. 量价关系验证

## 一、模块定位

量价关系分析模块位于 **D. 股价形态判断层**，属于项目核心业务逻辑，放在 `modules/volume_price/` 下。与蔡森形态识别模块并列，提供三类核心能力：
1. **基础量价关系识别**：识别每日的量价关系类型和强度
2. **高级量价模式检测**：识别量窒息、底量超顶量、量价背离等关键信号
3. **量价验证融合**：为蔡森形态提供量价验证评分

所有分析结果通过 `lib/db/` 持久化到 MongoDB。

**依赖关系**：
- K线数据输入：来自 `lib/data_source/` 的 DataFrame
- 蔡森形态数据结构：来自 `modules/caisen/pattern_base.py`
- 缓存读写：通过 `lib/db/cache_manager.py` 操作 MongoDB
- 交易信号输出：供 `modules/analysis/signal_generator.py` 使用

## 二、数据流

```
lib/data_source/data_fetcher.py
    ↓ DataFrame [trade_date, close, volume, ...]
modules/volume_price/basic_relation.py
    ↓ List[VolumePriceSignal] (每日基础量价信号)
modules/volume_price/advanced_pattern.py
    ↓ List[AdvancedSignal] (高级量价信号)
modules/volume_price/volume_verifier.py
    ↓ float (量价验证得分 0-100) → 供 modules/caisen/strength_scorer.py 调用
    ↓ 全部结果通过 lib/db/cache_manager.py 缓存到 MongoDB
```

## 三、基础量价关系识别

### 3.1 文件：`modules/volume_price/basic_relation.py`

### 3.2 四种基本量价关系

| 类型 | 代码 | 价格趋势 | 成交量趋势 | 含义 |
|------|------|----------|------------|------|
| 价升量增 | `price_up_vol_up` | close > MA5(close) | volume > MA5(volume) | 健康上涨 |
| 价升量缩 | `price_up_vol_down` | close > MA5(close) | volume < MA5(volume) | 上涨乏力 |
| 价跌量增 | `price_down_vol_up` | close < MA5(close) | volume > MA5(volume) | 恐慌下跌 |
| 价跌量缩 | `price_down_vol_down` | close < MA5(close) | volume < MA5(volume) | 抛压衰竭 |

### 3.3 数据结构
```python
from enum import Enum
from dataclasses import dataclass, field
from typing import List, Optional

class VolumePriceType(Enum):
    PRICE_UP_VOL_UP = "price_up_vol_up"       # 价升量增
    PRICE_UP_VOL_DOWN = "price_up_vol_down"   # 价升量缩
    PRICE_DOWN_VOL_UP = "price_down_vol_up"   # 价跌量增
    PRICE_DOWN_VOL_DOWN = "price_down_vol_down" # 价跌量缩
    NEUTRAL = "neutral"                        # 均线缠绕/无明确趋势

class AdvancedType(Enum):
    VOLUME_EXHAUST = "volume_exhaust"         # 量窒息（地量）
    BOTTOM_VOL_OVER_TOP = "bottom_vol_over_top" # 底量超顶量
    DIVERGENCE_TOP = "divergence_top"         # 顶背离（价创新高，量萎缩）
    DIVERGENCE_BOTTOM = "divergence_bottom"   # 底背离（价创新低，量萎缩）
    HIGH_VOL_STALL = "high_vol_stall"         # 高位放量滞涨
    BREAKOUT_VOL = "breakout_vol"             # 突破放量

@dataclass
class VolumePriceSignal:
    ts_code: str
    period: str
    trade_date: str
    relation_type: VolumePriceType
    strength: float              # 关系强度 (0.0 - 1.0)
    close: float
    volume: float
    ma5_price: float
    ma5_volume: float
    pct_chg: float               # 涨跌幅
    is_advanced: bool = False
    advanced_type: Optional[AdvancedType] = None
    advanced_detail: Optional[dict] = None
    update_time: str = ""
```

### 3.4 算法实现
```python
import numpy as np
import pandas as pd

class BasicRelationAnalyzer:

    def __init__(self, config: dict):
        self.config = config
        self.ma_period = config.get("volume_price", {}).get("ma_period", 5)

    def analyze(self, ts_code: str, period: str,
                kline_df: pd.DataFrame) -> List[VolumePriceSignal]:
        """
        分析每日量价关系
        """
        if kline_df.empty or len(kline_df) < self.ma_period:
            return []

        df = kline_df.copy()
        df = df.sort_values("trade_date").reset_index(drop=True)

        df["ma_price"] = df["close"].rolling(window=self.ma_period).mean()
        df["ma_volume"] = df["volume"].rolling(window=self.ma_period).mean()
        df = df.dropna(subset=["ma_price", "ma_volume"])

        signals = []
        for _, row in df.iterrows():
            signal = self._analyze_single_row(ts_code, period, row)
            signals.append(signal)

        return signals

    def _analyze_single_row(self, ts_code: str, period: str, row) -> VolumePriceSignal:
        close = row["close"]
        volume = row["volume"]
        ma_price = row["ma_price"]
        ma_volume = row["ma_volume"]

        price_up = close > ma_price
        price_down = close < ma_price
        vol_up = volume > ma_volume
        vol_down = volume < ma_volume

        if price_up and vol_up:
            relation = VolumePriceType.PRICE_UP_VOL_UP
        elif price_up and vol_down:
            relation = VolumePriceType.PRICE_UP_VOL_DOWN
        elif price_down and vol_up:
            relation = VolumePriceType.PRICE_DOWN_VOL_UP
        elif price_down and vol_down:
            relation = VolumePriceType.PRICE_DOWN_VOL_DOWN
        else:
            relation = VolumePriceType.NEUTRAL

        strength = self._calc_strength(close, ma_price, volume, ma_volume, relation)

        return VolumePriceSignal(
            ts_code=ts_code,
            period=period,
            trade_date=row["trade_date"],
            relation_type=relation,
            strength=strength,
            close=close,
            volume=volume,
            ma5_price=ma_price,
            ma5_volume=ma_volume,
            pct_chg=row.get("pct_chg", 0),
        )

    def _calc_strength(self, close, ma_price, volume, ma_volume,
                       relation: VolumePriceType) -> float:
        """
        计算量价关系强度
        强度 = 价格偏离度 × 成交量偏离度 的归一化值
        """
        if relation == VolumePriceType.NEUTRAL:
            return 0.0

        price_dev = abs(close - ma_price) / ma_price if ma_price > 0 else 0
        vol_dev = abs(volume - ma_volume) / ma_volume if ma_volume > 0 else 0

        strength = 0.5 * min(price_dev * 10, 1.0) + 0.5 * min(vol_dev * 5, 1.0)
        return round(min(max(strength, 0.0), 1.0), 4)
```

```python
class VolumeExhaustDetector:

    def __init__(self, config: dict):
        vp_config = config.get("volume_price", {})
        self.window = vp_config.get("volume_exhaust_window", 120)
        self.ratio = vp_config.get("volume_exhaust_ratio", 1.3)
        self.shrink_days = vp_config.get("volume_exhaust_shrink_days", 5)

    def detect(self, kline_df: pd.DataFrame) -> List[dict]:
        """
        检测量窒息信号
        条件:
        1. 当日成交量 < 120日最低成交量 × 1.3
        2. 连续5日成交量萎缩
        3. 价格处于底部区域（200日均线以下，或窄幅震荡）
        """
        if len(kline_df) < self.window:
            return []

        df = kline_df.sort_values("trade_date").reset_index(drop=True)
        signals = []

        for i in range(self.window, len(df)):
            row = df.iloc[i]
            window_df = df.iloc[i - self.window:i]

            min_vol = window_df["volume"].min()
            if row["volume"] > min_vol * self.ratio:
                continue

            if i < self.shrink_days:
                continue
            recent = df.iloc[i - self.shrink_days + 1:i + 1]
            if not all(recent["volume"].diff().dropna() < 0):
                shrink_count = (recent["volume"].diff().dropna() < 0).sum()
                if shrink_count < self.shrink_days - 1:
                    continue

            ma200 = window_df["close"].rolling(200).mean().iloc[-1]
            if not pd.isna(ma200) and row["close"] > ma200:
                continue

            exhaust_ratio = row["volume"] / min_vol

            signals.append({
                "trade_date": row["trade_date"],
                "advanced_type": AdvancedType.VOLUME_EXHAUST,
                "volume": row["volume"],
                "min_volume_120d": min_vol,
                "exhaust_ratio": exhaust_ratio,
                "close": row["close"],
                "ma200": ma200 if not pd.isna(ma200) else None,
                "detail": {
                    "exhaust_ratio": round(exhaust_ratio, 4),
                    "shrink_days": self.shrink_days,
                }
            })

        return signals
```

```python
class BottomVolOverTopDetector:

    def __init__(self, config: dict):
        self.min_ratio = config.get("volume_price", {}).get("bottom_volume_ratio", 0.80)

    def detect(self, kline_df: pd.DataFrame) -> List[dict]:
        """
        检测底量超顶量信号
        条件: 底部区域成交量达到或超过前期顶部成交量的 80%-100%
        """
        if len(kline_df) < 60:
            return []

        df = kline_df.sort_values("trade_date").reset_index(drop=True)
        signals = []

        for i in range(60, len(df)):
            window = df.iloc[i - 60:i]

            top_idx = window["high"].idxmax()
            top_area = window.iloc[max(0, top_idx - 5):top_idx + 5]
            top_avg_vol = top_area["volume"].mean()

            current_vol = df.iloc[i]["volume"]

            current_close = df.iloc[i]["close"]
            window_high = window["high"].max()
            if current_close > window_high * 0.7:
                continue

            if top_avg_vol > 0 and current_vol / top_avg_vol >= self.min_ratio:
                signals.append({
                    "trade_date": df.iloc[i]["trade_date"],
                    "advanced_type": AdvancedType.BOTTOM_VOL_OVER_TOP,
                    "current_volume": current_vol,
                    "top_avg_volume": top_avg_vol,
                    "ratio": current_vol / top_avg_vol,
                    "detail": {
                        "ratio": round(current_vol / top_avg_vol, 4),
                        "min_ratio": self.min_ratio,
                    }
                })

        return signals
```

### 4.4 量价背离检测

```python
class DivergenceDetector:

    def detect(self, kline_df: pd.DataFrame, lookback: int = 20) -> List[dict]:
        """
        检测量价背离信号

        顶背离: 价格创新高，但成交量未创新高（甚至萎缩）
        底背离: 价格创新低，但成交量未创新低（甚至萎缩）
        """
        if len(kline_df) < lookback * 2:
            return []

        df = kline_df.sort_values("trade_date").reset_index(drop=True)
        signals = []

        for i in range(lookback, len(df)):
            window = df.iloc[i - lookback:i + 1]
            current = df.iloc[i]

            max_price_idx = window["high"].idxmax()
            min_price_idx = window["low"].idxmax()

            if current["high"] >= window["high"].iloc[:-1].max():
                prev_high_vol = window["volume"].iloc[
                    window["high"] == window["high"].iloc[:-1].max()
                ].iloc[0] if (window["high"][:-1] == window["high"].iloc[:-1].max()).any() else 0

                if prev_high_vol > 0 and current["volume"] < prev_high_vol:
                    signals.append({
                        "trade_date": current["trade_date"],
                        "advanced_type": AdvancedType.DIVERGENCE_TOP,
                        "current_high": current["high"],
                        "current_volume": current["volume"],
                        "prev_high_volume": prev_high_vol,
                        "detail": {
                            "divergence_ratio": round(current["volume"] / prev_high_vol, 4),
                        }
                    })

            if current["low"] <= window["low"].iloc[:-1].min():
                prev_low_vol = window["volume"].iloc[
                    window["low"] == window["low"].iloc[:-1].min()
                ].iloc[0] if (window["low"][:-1] == window["low"].iloc[:-1].min()).any() else 0

                if prev_low_vol > 0 and current["volume"] < prev_low_vol:
                    signals.append({
                        "trade_date": current["trade_date"],
                        "advanced_type": AdvancedType.DIVERGENCE_BOTTOM,
                        "current_low": current["low"],
                        "current_volume": current["volume"],
                        "prev_low_volume": prev_low_vol,
                        "detail": {
                            "divergence_ratio": round(current["volume"] / prev_low_vol, 4),
                        }
                    })

        return signals
```

```python
class HighVolStallDetector:

    def detect(self, kline_df: pd.DataFrame, lookback: int = 10) -> List[dict]:
        """
        高位放量滞涨: 成交量放大但价格涨幅很小
        条件:
        1. 成交量 > 5日均量的 2倍
        2. 涨跌幅 < 1%（绝对值）
        3. 价格处于近期高位（20日内最高价的 90% 以上）
        """
        if len(kline_df) < 20:
            return []

        df = kline_df.sort_values("trade_date").reset_index(drop=True)
        df["ma5_vol"] = df["volume"].rolling(5).mean()
        signals = []

        for i in range(20, len(df)):
            row = df.iloc[i]
            window = df.iloc[i - 20:i]

            if pd.isna(row["ma5_vol"]) or row["volume"] < row["ma5_vol"] * 2:
                continue

            pct = row.get("pct_chg", 0) or 0
            if abs(pct) > 1.0:
                continue

            recent_high = window["high"].max()
            if row["close"] < recent_high * 0.9:
                continue

            signals.append({
                "trade_date": row["trade_date"],
                "advanced_type": AdvancedType.HIGH_VOL_STALL,
                "volume": row["volume"],
                "ma5_volume": row["ma5_vol"],
                "vol_ratio": row["volume"] / row["ma5_vol"],
                "pct_chg": pct,
                "close": row["close"],
                "recent_high": recent_high,
                "detail": {
                    "vol_ratio": round(row["volume"] / row["ma5_vol"], 4),
                    "pct_chg": pct,
                }
            })

        return signals
```

```python
class AdvancedPatternDetector:

    def __init__(self, config: dict):
        self.config = config
        self.detectors = {
            "volume_exhaust": VolumeExhaustDetector(config),
            "bottom_vol_over_top": BottomVolOverTopDetector(config),
            "divergence": DivergenceDetector(),
            "high_vol_stall": HighVolStallDetector(),
        }

    def detect(self, kline_df: pd.DataFrame) -> List[dict]:
        """
        运行所有高级模式检测器

        all_signals = []

            signals = detector.detect(kline_df)
            all_signals.extend(signals)
        return all_signals
```

## 五、量价验证（与蔡森形态融合）

### 5.1 文件：`modules/volume_price/volume_verifier.py`

### 5.2 验证机制

为每种蔡森形态定义量价验证规则，采用加权评分体系。本模块被 `modules/caisen/strength_scorer.py` 调用。

```python
from modules.caisen.pattern_base import PatternResult, PatternType, PatternStatus

class VolumeVerifier:
    """
    量价验证器
    为蔡森形态提供量价验证评分 (0-100)
    被 modules/caisen/strength_scorer.py 调用
    """

    def __init__(self, config: dict):
        self.config = config
        self.breakout_ratio = config.get("volume_price", {}).get("volume_breakout_ratio", 1.5)

    def verify(self, pattern: PatternResult, kline_df: pd.DataFrame) -> float:
        """
        根据形态类型调用对应的验证逻辑
        """
        verifier = {
            PatternType.W_BOTTOM: self._verify_w_bottom,
            PatternType.M_TOP: self._verify_m_top,
            PatternType.HEAD_SHOULDER_BOTTOM: self._verify_hs_bottom,
            PatternType.HEAD_SHOULDER_TOP: self._verify_hs_top,
            PatternType.FLAG_DOWN: self._verify_flag_down,
            PatternType.FLAG_UP: self._verify_flag_up,
            PatternType.TRIANGLE_BOTTOM: self._verify_triangle_bottom,
            PatternType.TRIANGLE_HEAD: self._verify_triangle_head,
            PatternType.BREAKOUT_FAIL: self._verify_breakout_fail,
            PatternType.PODIE_FAN: self._verify_podie_fan,
        }

        func = verifier.get(pattern.pattern_type)
        if func:
            return func(pattern, kline_df)
        return 0.0

    def _verify_w_bottom(self, pattern: PatternResult, kline_df: pd.DataFrame) -> float:
        """
        W底量价验证 (100分)
        - 右底缩量: 25分
        - 突破放量: 35分
        - 底量超顶量: 25分
        - 突破后持续放量: 15分
        """
        score = 0.0

        left_bottom_date = pattern.key_points.get("left_bottom")
        right_bottom_date = pattern.key_points.get("right_bottom")
        if left_bottom_date and right_bottom_date:
            left_vol = self._avg_volume(kline_df, left_bottom_date.date, 5)
            right_vol = self._avg_volume(kline_df, right_bottom_date.date, 5)
            if left_vol > 0 and right_vol < left_vol:
                shrink_ratio = right_vol / left_vol
                score += 25 * (1 - shrink_ratio)

        breakout = pattern.key_points.get("breakout")
        if breakout and pattern.status == PatternStatus.CONFIRMED:
            breakout_vol = self._get_volume(kline_df, breakout.date)
            ma5_vol = self._avg_volume(kline_df, breakout.date, 5)
            if ma5_vol > 0:
                vol_ratio = breakout_vol / ma5_vol
                if vol_ratio >= self.breakout_ratio:
                    score += 35
                elif vol_ratio >= 1.2:
                    score += 35 * (vol_ratio - 1.2) / (self.breakout_ratio - 1.2)

        neckline = pattern.key_points.get("neckline_left")
        if neckline and left_bottom_date and right_bottom_date:
            top_vol = self._avg_volume(kline_df, neckline.date, 5)
            bottom_vol = max(
                self._avg_volume(kline_df, left_bottom_date.date, 5),
                self._avg_volume(kline_df, right_bottom_date.date, 5)
            )
            if top_vol > 0:
                ratio = bottom_vol / top_vol
                if ratio >= 1.0:
                    score += 25
                elif ratio >= 0.8:
                    score += 25 * (ratio - 0.8) / 0.2

        if breakout and pattern.status == PatternStatus.CONFIRMED:
            post_vol = self._avg_volume_after(kline_df, breakout.date, 3)
            pre_vol = self._avg_volume_before(kline_df, breakout.date, 5)
            if pre_vol > 0 and post_vol > pre_vol:
                score += 15

        return round(min(score, 100.0), 2)

    def _verify_m_top(self, pattern: PatternResult, kline_df: pd.DataFrame) -> float:
        """
        M头量价验证 (100分)
        - 右顶缩量: 25分
        - 跌破放量: 35分
        - 天量见天价: 25分
        - 跌破后持续放量: 15分
        """
        score = 0.0

        left_top = pattern.key_points.get("left_top")
        right_top = pattern.key_points.get("right_top")
        if left_top and right_top:
            left_vol = self._avg_volume(kline_df, left_top.date, 5)
            right_vol = self._avg_volume(kline_df, right_top.date, 5)
            if left_vol > 0 and right_vol < left_vol:
                score += 25 * (1 - right_vol / left_vol)

        breakdown = pattern.key_points.get("breakdown")
        if breakdown and pattern.status == PatternStatus.CONFIRMED:
            bd_vol = self._get_volume(kline_df, breakdown.date)
            ma5_vol = self._avg_volume(kline_df, breakdown.date, 5)
            if ma5_vol > 0 and bd_vol / ma5_vol >= self.breakout_ratio:
                score += 35

        if left_top and right_top:
            top_vol = max(
                self._avg_volume(kline_df, left_top.date, 5),
                self._avg_volume(kline_df, right_top.date, 5)
            )
            start_date = pattern.start_date
            pre_vol = self._avg_volume(kline_df, start_date, 20)
            if pre_vol > 0 and top_vol / pre_vol >= 2.0:
                score += 25

        if breakdown and pattern.status == PatternStatus.CONFIRMED:
            post_vol = self._avg_volume_after(kline_df, breakdown.date, 3)
            pre_vol = self._avg_volume_before(kline_df, breakdown.date, 5)
            if pre_vol > 0 and post_vol > pre_vol:
                score += 15

        return round(min(score, 100.0), 2)

    def _verify_hs_bottom(self, pattern: PatternResult, kline_df: pd.DataFrame) -> float:
        """
        头肩底量价验证 (100分)
        - 头部缩量: 20分
        - 右肩缩量: 20分
        - 突破放量: 35分
        - 底量超顶量: 25分
        """
        score = 0.0
        left_shoulder = pattern.key_points.get("left_shoulder")
        head = pattern.key_points.get("head")
        right_shoulder = pattern.key_points.get("right_shoulder")
        neckline = pattern.key_points.get("neckline")

        if left_shoulder and head:
            ls_vol = self._avg_volume(kline_df, left_shoulder.date, 5)
            head_vol = self._avg_volume(kline_df, head.date, 5)
            if ls_vol > 0 and head_vol < ls_vol:
                score += 20 * (1 - head_vol / ls_vol)

        if right_shoulder and head:
            rs_vol = self._avg_volume(kline_df, right_shoulder.date, 5)
            head_vol = self._avg_volume(kline_df, head.date, 5)
            if head_vol > 0 and rs_vol < head_vol:
                score += 20 * (1 - rs_vol / head_vol)

        breakout = pattern.key_points.get("breakout")
        if breakout and pattern.status == PatternStatus.CONFIRMED:
            vol = self._get_volume(kline_df, breakout.date)
            ma5 = self._avg_volume(kline_df, breakout.date, 5)
            if ma5 > 0 and vol / ma5 >= self.breakout_ratio:
                score += 35

        return round(min(score, 100.0), 2)

    def _verify_hs_top(self, pattern: PatternResult, kline_df: pd.DataFrame) -> float:
        """
        头肩顶量价验证 (100分) — 头肩底对称
        """
        score = 0.0
        # 对称逻辑: 头部天量、右肩缩量、跌破放量
        return round(min(score, 100.0), 2)
```

### 5.3 工具方法
```python
def _avg_volume(self, kline_df: pd.DataFrame, trade_date: str, window: int) -> float:
    """获取指定日期前后 window/2 天的平均成交量"""
    mask = kline_df["trade_date"] == trade_date
    if not mask.any():
        return 0.0
    idx = kline_df[mask].index[0]
    start = max(0, idx - window // 2)
    end = min(len(kline_df), idx + window // 2 + 1)
    return kline_df.iloc[start:end]["volume"].mean()

def _get_volume(self, kline_df: pd.DataFrame, trade_date: str) -> float:
    """获取指定日期的成交量"""
    mask = kline_df["trade_date"] == trade_date
    if mask.any():
        return kline_df[mask].iloc[0]["volume"]
    return 0.0

def _avg_volume_after(self, kline_df: pd.DataFrame, trade_date: str, days: int) -> float:
    """获取指定日期之后 days 天的平均成交量"""
    mask = kline_df["trade_date"] == trade_date
    if not mask.any():
        return 0.0
    idx = kline_df[mask].index[0]
    end = min(len(kline_df), idx + days + 1)
    return kline_df.iloc[idx + 1:end]["volume"].mean()

def _avg_volume_before(self, kline_df: pd.DataFrame, trade_date: str, days: int) -> float:
    """获取指定日期之前 days 天的平均成交量"""
    mask = kline_df["trade_date"] == trade_date
    if not mask.any():
        return 0.0
    idx = kline_df[mask].index[0]
    start = max(0, idx - days)
    return kline_df.iloc[start:idx]["volume"].mean()
```

### 6.2 信号生成器

信号生成器放在 `modules/analysis/signal_generator.py`，综合 `modules/caisen/` 和 `modules/volume_price/` 的结果：

```python
from modules.caisen.pattern_base import PatternType, PatternStatus, PatternDirection, PatternResult
from modules.volume_price.basic_relation import VolumePriceType, AdvancedType, VolumePriceSignal

class SignalGenerator:
    """
    交易信号生成器
    综合形态识别和量价分析结果生成交易信号
    """

    def __init__(self, config: dict):
        self.config = config
        self.min_strength = config.get("caisen", {}).get("signal_strength_threshold", 70)
        self.min_volume = config.get("caisen", {}).get("signal_volume_threshold", 60)

    def generate(self, patterns: List[PatternResult],
                 vp_signals: List[VolumePriceSignal],
                 advanced_signals: List[dict]) -> List[TradeSignal]:
        """
        生成交易信号
        """
        signals = []

        for pattern in patterns:
            if pattern.status != PatternStatus.CONFIRMED:
                continue
            if pattern.strength_score < self.min_strength:
                continue
            if pattern.volume_score < self.min_volume:
                continue

            signal = self._pattern_to_signal(pattern)
            signals.append(signal)

        for adv in advanced_signals:
            adv_type = adv.get("advanced_type")
            if adv_type == AdvancedType.DIVERGENCE_TOP:
                signals.append(self._divergence_warning(adv, SignalType.REDUCE))
            elif adv_type == AdvancedType.DIVERGENCE_BOTTOM:
                signals.append(self._divergence_warning(adv, SignalType.WARNING))
            elif adv_type == AdvancedType.HIGH_VOL_STALL:
                signals.append(self._stall_warning(adv))

        exhaust_signals = [s for s in advanced_signals
                          if s.get("advanced_type") == AdvancedType.VOLUME_EXHAUST]
        for es in exhaust_signals:
            matching_vp = [v for v in vp_signals
                          if v.trade_date >= es["trade_date"]
                          and v.relation_type == VolumePriceType.PRICE_UP_VOL_UP]
            if matching_vp:
                signals.append(self._exhaust_buy_signal(es, matching_vp[0]))

        return signals

    def _pattern_to_signal(self, pattern: PatternResult) -> TradeSignal:
        reasons = []
        if pattern.strength_score >= 80:
            reasons.append(f"形态强弱评分 {pattern.strength_score} 分(优秀)")
        if pattern.volume_score >= 70:
            reasons.append(f"量价验证 {pattern.volume_score} 分(充分)")

        signal_type = SignalType.BUY if pattern.direction == PatternDirection.BULLISH else SignalType.SELL

        return TradeSignal(
            ts_code=pattern.ts_code,
            period=pattern.period,
            signal_type=signal_type,
            signal_date=pattern.end_date,
            pattern_type=pattern.pattern_type,
            entry_price=pattern.neckline_price,
            stop_loss_price=pattern.stop_loss_price or 0,
            target_price=pattern.target_price,
            strength_score=pattern.strength_score,
            volume_score=pattern.volume_score,
            confidence=pattern.confidence,
            source=f"蔡森{pattern.pattern_type.value}形态",
            reasons=reasons,
        )
```

## 七、缓存策略

### 7.1 基础量价信号缓存
- Collection: `volume_price_signals`
- 唯一键: `{ts_code, period, trade_date}`
- 通过 `lib/db/cache_manager.py` 进行缓存读写
- 缓存策略: 优先读取，未命中则计算并写入
- 增量更新: 删除最新缓存日期之后的数据，重新计算

### 7.2 高级量价信号缓存
- 高级信号作为 `volume_price_signals` 中的 `is_advanced=true` 记录存储
- 无需单独 collection

### 7.3 交易信号缓存
- Collection: `trade_signals`
- 唯一键: `{ts_code, period, signal_date, pattern_type}`
- 每次重新生成（基于最新的形态和量价数据）
